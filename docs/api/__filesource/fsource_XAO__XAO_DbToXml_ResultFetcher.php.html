13    * Results fetcher for various Database Results classes
14    *
15    * There are a number of class libraries for querying databases. This class is
16    * designed to provide wrappers to most of those with the express purpose of
17    * getting an array of associative arrays. It's this 2D array that DbToXml needs
18    * to produce XML. This class is only instantiated if such an array is not passed
19    * to the DbToXml constructor in the first place. This class first checks to see
20    * which (if any) type of DbResult class is being used, then it calls a dedicated
21    * method to extract the 2D array. Note that we choose not to support native
22    * database result resources since we are not developing/supporting yet another
23    * database abstraction layer library. This is a just a little
24    * icing on te cake to eliminate the extra steps needed for obtaining, checking
25    * for errors, and formatting the actual data into an array of associative
26    * arrays. It means that users of database abstraction layer classes need not
27    * concern themselves with the format that DbToXml expects.
28    *
29    * @author       Terence Kearns
30    * @version      1.0
31    * @copyright    Terence Kearns 2003
32    * @license      LGPL
33    * @package      XAO
34    * @link         http://xao-php.sourceforge.net
35    */

42        function ResultFetcher(&$objResult) {
43            if(is_object($objResult)) {
44                $this->objResult =& $objResult;
45            }
46            else {
47                $this->Throw(
48                    "ResultFetcher: Only [result] objects are supported."
49                    ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
50                );
51                return;
52            }
53            
54                                            // PEAR DB
55            if(is_a($objResult,"DB_Result")) {
56                $this->MkPearResult();
57            }
58            // insert more elseif() tests (more supported objects)
59            else {
60                $this->Throw(
61                    "ResultFetcher: The DB result object you passed to DbToXml "
62                    ."returned an error:\n".$objDBResult->getMessage() 
63                    //.":\n".$objDBResult->getUserinfo()
64                    ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
65                );
66            }
67        }
68        
69        function &arrGetResult() {
70            if(is_array($this->arr2D)) return $this->arr2D;
71            return array();
72        }
73        
74        function MkPearResult() {
75            if(DB::isError($this->objResult)) {
76                $this->Throw(
77                    "ResultFetcher: The DB result object you passed to DbToXml "
78                    ."returned an error:\n".$this->objResult->getMessage() 
79                    //.":\n".$objDBResult->getUserinfo() // security issue
80                    ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
81                );
82            }
83            else {
84                                            // fetch the result data (list of rows) 
85                                            // into an associative array.
86                while($this->arr2D[] = $this->objResult->fetchRow(DB_FETCHMODE_ASSOC));
87                array_pop($this->arr2D);
88            }
89        }
