137       function DbToXml(&$mxdResult,$arrCallBacks = array(),$objDocRef = null,$ndStub = null) {
138           $this->arrCallBacks = $arrCallBacks;
139           foreach($this->arrCallBacks AS $fncName) {
140                   if(!method_exists($this,$fncName)) {
141                   $this->Throw(
142                       "     DbToXml: Call-backfunction$fncNamedoesnotexist."
143                       ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
144                   );
145                   return;
146                   }
147           }
148                                           // run the parent constructor
149           if(is_object($objDocRef) && is_object($ndStub)) {
150                                           // check to see if this tree is being
151                                           // grafted onto an existing node.
152               $this->DomDoc($objDocRef,XAO_DOC_REFERENCE);
153               $this->ndStub = $ndStub;
154           }
155           else {
156               $this->DomDoc($this->strResEl);
157               $this->ndStub = $this->ndRoot;
158           }
159                                           // populate the 2d associative array 
160                                           // that will be used to build our XML
161           if(is_array($mxdResult)) {
162               $this->arrResult = $mxdResult;
163           }
164           elseif(is_object($mxdResult)) {
165               $objFetcher = new ResultFetcher($mxdResult);
166               if(strlen($objFetcher->strError)) {
167                   $this->Throw(
168                       "DbToXml: ".$objFetcher->strError
169                       ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
170                   );
171               }
172               else {
173                   $this->arrResult =& $objFetcher->arrGetResult();
174               }
175           }
176           else {
177               $this->Throw(
178                   "DbToXml: Constructor expects a 2D array (or a supported result 
179                   object)."
180                   ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
181               );
182           }
183       }

211       /**
212       * Convert the RDBMS data into XML elements.
213       *
214       * Iterate over $this->arrResult to first obtain the rows and then the
215       * fields which are created and appended as elements using DOM XML methods.
216       *
217       * @return   void 
218       * @access   public
219       */
220       function Execute() {
221               if(!is_array($this->arrResult)) {
222               $this->Throw(
223                   "DbToXml: For some reason, the result set is not formatted 
224                   [as an] array." ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
225               );
226               return;
227               }
228               
229               if(is_object($this->ndStub)) {
230                       if(!$this->blnTestElementNode($this->ndStub)) {
231                               $this->Throw(
232                       "DbToXml: The stub node is not a valid domelement."
233                       ,$this->arrSetErrFnc(__FUNCTION__,__LINE__)
234                   );
235                   return;
236                       }
237               }
238                                           // iterate through the result list
239           foreach($this->arrResult AS $arrRow) {
240                                           // add a row element for each row in the 
241                                           // result list
242               $elRow = $this->objDoc->create_element($this->strRowEl);
243               $ndRow = $this->ndStub->append_child($elRow);
244               $this->RowConstructor($ndRow);
245                                           // iterate through the fields in the row
246               foreach($arrRow AS $fieldName => $fieldVal) {
247                                           // DON'T CREATE EMPTY TAGS!
248                   if(strlen($fieldVal) && !is_int($fieldName)) {
249                                           // add an element for each non-empty field
250                       $elField = $this->objDoc->create_element($fieldName);
251                       $ndField = $ndRow->append_child($elField);
252                                           // CHECK FOR CALLBACKS
253                       if(isset($this->arrCallBacks[$fieldName])) {
254                           $funcName = $this->arrCallBacks[$fieldName];
255                           $this->$funcName($this->objDoc,$ndField,$fieldVal);
256                       }
257                       else {
258                           $ndField->set_content($fieldVal);
259                       }
260                   }
261               }
262               $this->RowDestructor($ndRow);
263           }
264       }

325       * unix timestamp date call-back mutator function
326       *
327       * This call-back function is placed here for convenience and can be used to
328       * produce an element with a more convenient schema for representing a date
329       * from a unix timestamp. It also provides a useful example of the call-back
330       * capability of this class. What it does is add a set of attributes to the
331       * element in context which each represent a conventional date component.
332       *
333       * @param    object  a reference to the current PHP DOM XML object instance
334       * @param    node    a reference to the element node representing a field
335       * @param    string  a copy of the text content that would normally be
336       *                    assigned to this field elemeent.
337       * @access   public
338       * @return   void 
339       */
340       function unixTsToReadable(&$ndField,$intTs) {
341           $intTs = (integer)$intTs;
342           if($intTs < 0) return;
343           $ndField->set_attribute("unixTS",$intTs);
344           $ndField->set_attribute("ODBCformat",date("Y-m-d H:i:s",$intTs));
345           $ndField->set_attribute("year",date("Y",$intTs));
346           $ndField->set_attribute("month",date("m",$intTs));
347           $ndField->set_attribute("day",date("d",$intTs));
348           $ndField->set_attribute("hour",date("H",$intTs));
349           $ndField->set_attribute("min",date("i",$intTs));
350       }

352       /**
353       * ODBC timestamp date call-back mutator function
354       *
355       * This call-back function is placed here for convenience and can be used to
356       * produce create a unix timestamp from an ODBC compliant timestamp. This
357       * unix timestap is then sent to $this->unixTsToReadable where nice
358       * attributes are added :)
359       *
360       * @param    object  a reference to the current PHP DOM XML object instance
361       * @param    node    a reference to the element node representing a field
362       * @param    string  a copy of the text content that would normally be
363       *                    assigned to this field elemeent.
364       * @access   public
365       * @return   void 
366       */
367       function odbcToReadable(&$ndField,$odbcTs) {
368           if(trim($odbcTs) == "") return;
369           // example of MS SQL select snippet producing expected format.
370           // CAST(DATEPART(yyyy,myDate) AS varchar(64))    + '-' +
371           // CAST(DATEPART(mm,myDate) AS varchar(64))      + '-' +
372           // CAST(DATEPART(dd,myDate) AS varchar(64))      + ' ' +
373           // CAST(DATEPART(hh,myDate) AS varchar(64))      + ':' +
374           // CAST(DATEPART(n,myDate) AS varchar(64)) AS myODBCDate,
375           $arrTs = explode(" ",$odbcTs);
376           $arrDate = explode("-",$arrTs[0]);
377           $arrTime = explode(":",$arrTs[1]);
378           $year   = $arrDate[0];
379           $month  = $arrDate[1];
380           $day    = $arrDate[2];
381           $hour   = $arrTime[0];
382           $min    = $arrTime[1];
383           if($year != 1900) {
384               $year = date("Y");
385               $unixTs = mktime($hour,$min,0,$month,$day,$year);
386               if($unixTs != -1) $this->unixTsToReadable($ndField,$unixTs);
387               return $odbcTs;
388           }
389       }
